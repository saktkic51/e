local function safeTeleport(destination)
    if not root or not root.Parent then
        return
    end
    if tpConnection then
        tpConnection:Disconnect()
        tpConnection = nil
    end
    tpConnection = RunService.Heartbeat:Connect(function()
        local currentPos = root.Position
        root:PivotTo(ventCFrame)
        if (currentPos - targetPos).Magnitude <= distance then
            root:PivotTo(destination)
            tpConnection:Disconnect()
            tpConnection = nil
        end
    end)
end

local function refreshPlayers()
    for _, child in ipairs(playersScrollingFrame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    local numBtns = 0
    for _, targetPlr in ipairs(Players:GetPlayers()) do
        if targetPlr ~= plr then
            numBtns = numBtns + 1
            local btn = Instance.new("TextButton")
            btn.Name = targetPlr.Name
            btn.Size = UDim2.new(1, -12, 0, 40)
            btn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
            btn.TextColor3 = Color3.new(1, 1, 1)
            btn.Text = targetPlr.DisplayName
            btn.TextScaled = true
            btn.Parent = playersScrollingFrame
            btn.MouseButton1Click:Connect(function()
                local char = targetPlr.Character
                if char then
                    local hrp = char:FindFirstChild("HumanoidRootPart")
                    if hrp then
                        safeTeleport(hrp.CFrame)
                    else
                    end
                else
                end
            end)
        end
    end
    playersScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, numBtns * 45)
end
